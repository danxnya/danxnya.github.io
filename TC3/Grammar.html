  1. S -> Asignacion

  5. Factor -> ( Expresion ) | Numero | Identificador
  6. Numero -> Entero | Hexadecimal | NotacionCientifica
  7. Identificador -> Letra ( Letra | Digito )*
  8. Hexadecimal -> 0x ( Digito | A ... F )*
  9. NotacionCientifica -> Digito+ . Digito+ ( E | e ) [ + | - ] Digito+
  10. Entero -> Digito+
  11. DigitoHexadecimal -> 0 ... 9 | A ... F
